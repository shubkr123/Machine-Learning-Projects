def correlation(x,threshold):
    corr_column=set()
    corr_matrix=x.corr()
    for i in range(len(corr_matrix.columns)):
        for j in range(i):
            if abs(corr_matrix.iloc[i,j])>threshold:
                colname=corr_matrix.columns[i]
                corr_column.add(colname)
    return corr_column
